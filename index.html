        // 递归获取所有markdown文件

        async function getAllMarkdownFiles(dirHandle, path = '') {

            const files = [];

            

            for await (const entry of dirHandle.values()) {

                const entryPath = path ? `${path}/${entry.name}` : entry.name;

                

                if (entry.kind === 'file' && entry.name.endsWith('.md')) {

                    files.push({

                        handle: entry,

                        path: entryPath

                    });

                } else if (entry.kind === 'directory' && !entry.name.startsWith('.')) {

                    try {

                        const subDirHandle = await dirHandle.getDirectoryHandle(entry.name);

                        const subFiles = await getAllMarkdownFiles(subDirHandle, entryPath);

                        files.push(...subFiles);

                    } catch (error) {

                        console.warn(`无法访问目录 ${entryPath}:`, error);

                    }

                }

            }

            

            return files;

        }


        // 解析任务行

        function parseTaskLine(line, lineNumber, filePath) {

            // Strip carriage return for Windows line endings

            const cleanLine = line.replace(/\r$/, '');

            const taskRegex = /^- \[([ x\*\-\/!i?\"])\]\s+(.+?)(?:\s+📅\s*(\d{4}-\d{2}-\d{2}))?(?:\s+⏳\s*(\d{4}-\d{2}-\d{2}))?\s*$/;

            const match = cleanLine.match(taskRegex);

            

            if (!match) return null;

            

            const [, status, content, dueDate, scheduledDate] = match;

            

            // 提取标签（支持层级标签如 #s/events/finance）

            const tagRegex = /#([\w\/]+)/g;

            const tags = [];

            let tagMatch;

            while ((tagMatch = tagRegex.exec(content)) !== null) {

                const fullTag = '#' + tagMatch[1];

                tags.push(fullTag);

            }

            

            // 移除标签和日期后的纯文本

            const cleanContent = content.replace(/#[\w\/]+/g, '').replace(/\s+📅\s*\d{4}-\d{2}-\d{2}/g, '').replace(/\s+⏳\s*\d{4}-\d{2}-\d{2}/g, '').trim();

            

            return {

                file: filePath,

                lineNumber,

                status,

                text: cleanContent,

                tags,

                due: dueDate || null,

                scheduled: scheduledDate || null,

                subItems: [],

                rawLine: cleanLine

            };

        }


        // 加载所有任务 (内部函数)

        async function loadTasksInternal() {

            showMessage('正在加载任务...', 'info');

            

            try {

                const mdFiles = await getAllMarkdownFiles(directoryHandle);

                allTasks = [];

                

                for (const file of mdFiles) {

                    try {

                        const fileHandle = file.handle;

                        const fileContent = await fileHandle.getFile();

                        const text = await fileContent.text();

                        const lines = text.split('\n');

                        

                        for (let index = 0; index < lines.length; index++) {

                            const task = parseTaskLine(lines[index], index + 1, file.path);

                            if (task) {

                                // Capture indented sub-items

                                const subItems = [];

                                let subIndex = index + 1;

                                while (subIndex < lines.length) {

                                    const subLine = lines[subIndex].replace(/\r$/, ''); // Strip carriage return

                                    // Check if line is indented (starts with tab or spaces) and not a task

                                    if (/^[\t ]/.test(subLine) && !/^- \[/.test(subLine.trim())) {

                                        subItems.push(subLine.trim().replace(/^[-\t ]+/, ''));

                                        subIndex++;

                                    } else if (subLine.trim() === '') {

                                        subIndex++;

                                    } else {

                                        break;

                                    }

                                }

                                task.subItems = subItems;

                                allTasks.push(task);

                            }

                        }

                        

                        fileHandles.set(file.path, fileHandle);

                    } catch (error) {

                        console.warn(`无法读取文件 ${file.path}:`, error);

                    }

                }

                

                filteredTasks = [...allTasks];

                updateStats();

                updateTagCloud();

                filterTasks();

                showMessage(`成功加载 ${allTasks.length} 个任务`, 'success');

            } catch (error) {

                showMessage('加载失败: ' + error.message, 'error');

            }

        }

        // 更新标签云

        function updateTagCloud() {

            const tagCount = {};

            allTasks.forEach(task => {

                task.tags.forEach(tag => {

                    tagCount[tag] = (tagCount[tag] || 0) + 1;

                });

            });


            const tagCloud = document.getElementById('tagCloud');

            tagCloud.innerHTML = '';

            

            // 构建标签层次结构

            const tagHierarchy = {};

            Object.entries(tagCount).forEach(([tag, count]) => {

                const parts = tag.substring(1).split('/'); // 移除 # 并分割

                let current = tagHierarchy;

                parts.forEach((part, index) => {

                    const currentTag = '#' + parts.slice(0, index + 1).join('/');

                    if (!current[currentTag]) {

                        current[currentTag] = {

                            count: 0,

                            children: {},

                            level: index,

                            name: currentTag

                        };

                    }

                    current[currentTag].count += (index === parts.length - 1) ? count : 0;

                    current = current[currentTag].children;

                });

            });


            // 递归渲染标签层次

            function renderTagLevel(hierarchy, container, level = 0) {

                const sortedTags = Object.entries(hierarchy)

                    .sort((a, b) => {

                        // 先按层级排序，再按数量排序

                        if (a[1].level !== b[1].level) return a[1].level - b[1].level;

                        return b[1].count - a[1].count;

                    });


                sortedTags.forEach(([tag, data]) => {

                    const tagEl = document.createElement('div');

                    tagEl.style.marginLeft = `${data.level * 12}px`;

                    tagEl.style.marginBottom = '4px';

                    

                    const tagPill = document.createElement('span');

                    tagPill.className = 'tag-pill';

                    tagPill.style.fontSize = `${14 - data.level * 1}px`;

                    tagPill.style.opacity = `${1 - data.level * 0.1}`;

                    tagPill.textContent = `${tag} (${data.count})`;

                    tagPill.onclick = () => toggleTag(tag);

                    

                    tagEl.appendChild(tagPill);

                    container.appendChild(tagEl);

                    

                    // 递归渲染子标签

                    if (Object.keys(data.children).length > 0) {

                        renderTagLevel(data.children, container, level + 1);

                    }

                });

            }


            renderTagLevel(tagHierarchy, tagCloud);

        }

        // 筛选任务

        function filterTasks() {

            const searchTerm = document.getElementById('searchInput').value.toLowerCase();

            const fileDateFilter = document.getElementById('fileDateFilter').value;

            const dueFilter = document.getElementById('dueFilter').value;

            const sortBy = document.getElementById('sortBy').value;

            const statusCheckboxes = document.querySelectorAll('.checkbox-group input[type="checkbox"]:checked');

            const allowedStatuses = Array.from(statusCheckboxes).map(cb => cb.value);


            filteredTasks = allTasks.filter(task => {

                // 搜索筛选

                if (searchTerm && !task.text.toLowerCase().includes(searchTerm) &&

                    !task.tags.some(tag => tag.toLowerCase().includes(searchTerm)) &&

                    !task.file.toLowerCase().includes(searchTerm)) {

                    return false;

                }


                // 状态筛选

                if (!allowedStatuses.includes(task.status)) {

                    return false;

                }


                // 文件日期筛选

                if (fileDateFilter !== 'all') {

                    const today = new Date();

                    today.setHours(0, 0, 0, 0);

                    

                    switch (fileDateFilter) {

                        case 'today':

                            if (!isFileInDate(task.file, today)) return false;

                            break;

                        case 'week':

                            if (!isFileInWeek(task.file, today)) return false;

                            break;

                        case 'lastweek':

                            if (!isFileInLastWeek(task.file, today)) return false;

                            break;

                        case 'month':

                            if (!isFileInMonth(task.file, today)) return false;

                            break;

                    }

                }


                // 标签筛选（支持层级标签）

                const noTagsOnly = document.getElementById('noTagsOnly').checked;

                if (noTagsOnly) {

                    if (task.tags.length > 0) {

                        return false;

                    }

                } else if (selectedTags.size > 0) {

                    let hasMatchingTag = false;

                    const includeSubTags = document.getElementById('includeSubTags').checked;

                    

                    for (const selectedTag of selectedTags) {

                        // 检查是否直接匹配

                        if (task.tags.includes(selectedTag)) {

                            hasMatchingTag = true;

                            break;

                        }

                        

                        // 检查是否匹配子标签（仅在勾选"包含子标签"时）

                        if (includeSubTags) {

                            for (const taskTag of task.tags) {

                                if (taskTag.startsWith(selectedTag + '/')) {

                                    hasMatchingTag = true;

                                    break;

                                }

                            }

                        }

                        

                        if (hasMatchingTag) break;

                    }

                    

                    if (!hasMatchingTag) {

                        return false;

                    }

                }


                // 截止日期筛选 (包括 📅 截止日期和 ⏳ 计划日期)

                if (dueFilter === 'nodate') {

                    if (task.due || task.scheduled) return false;

                } else if (dueFilter !== 'all') {

                    // Get the effective date (due takes priority over scheduled)

                    const effectiveDate = task.due || task.scheduled;

                    if (!effectiveDate) return false;

                    

                    const taskDate = new Date(effectiveDate);

                    const today = new Date();

                    today.setHours(0, 0, 0, 0);

                    

                    switch (dueFilter) {

                        case 'overdue':

                            if (task.status === 'x') return false;

                            if (taskDate >= today) return false;

                            break;

                        case 'today':

                            if (!isToday(taskDate, today)) return false;

                            break;

                        case 'week':

                            if (!isInWeek(taskDate, today)) return false;

                            break;

                        case 'month':

                            if (!isInMonth(taskDate, today)) return false;

                            break;

                        case 'future':

                            if (taskDate <= today) return false;

                            break;

                    }

                }


                return true;

            });


            // 排序

            filteredTasks.sort((a, b) => {

                switch (sortBy) {

                    case 'due':

                        // Get effective date (due takes priority over scheduled)

                        const aEffective = a.due || a.scheduled;

                        const bEffective = b.due || b.scheduled;

                        if (!aEffective && !bEffective) return 0;

                        if (!aEffective) return 1;

                        if (!bEffective) return -1;

                        return new Date(aEffective) - new Date(bEffective);

                    case 'status':

                        return a.status.localeCompare(b.status);

                    case 'file':

                        return a.file.localeCompare(b.file);

                    case 'text':

                        return a.text.localeCompare(b.text);

                    case 'created':

                        return a.lineNumber - b.lineNumber;

                    default:

                        return 0;

                }

            });


            renderTasks();

        }

        function getDueClass(due, status) {

            if (!due || status === 'x') return '';

            

            const today = new Date();

            today.setHours(0, 0, 0, 0);

            if (dueDate < today) return 'overdue';

            if (isSameDay(dueDate, today)) return 'today';

            return '';

        }

        // 日期比较函数

        function isSameDay(date1, date2) {

            return date1.getFullYear() === date2.getFullYear() &&

                   date1.getMonth() === date2.getMonth() &&

                   date1.getDate() === date2.getDate();

        }


        function isInWeek(date, today) {

            const weekStart = new Date(today);

            weekStart.setDate(today.getDate() - today.getDay());

            const weekEnd = new Date(weekStart);

            weekEnd.setDate(weekStart.getDate() + 6);

            return date >= weekStart && date <= weekEnd;

        }


        function isInLastWeek(date, today) {

            const lastWeekStart = new Date(today);

            lastWeekStart.setDate(today.getDate() - today.getDay() - 7);

            const lastWeekEnd = new Date(lastWeekStart);

            lastWeekEnd.setDate(lastWeekStart.getDate() + 6);

            return date >= lastWeekStart && date <= lastWeekEnd;

        }


        function isFileInLastWeek(filePath, today) {

            // Extract date from file path (e.g., "Diary/2026/04-25.md")

            const dateMatch = filePath.match(/(\d{4})\/(\d{2})-(\d{2})\.md$/);

            if (!dateMatch) return false;

            

            const [, year, month, day] = dateMatch;

            const fileDate = new Date(parseInt(year), parseInt(month) - 1, parseInt(day));

            return isInLastWeek(fileDate, today);

        }


        function isFileInDate(filePath, today) {

            const [, year, month, day] = dateMatch;

            const fileDate = new Date(parseInt(year), parseInt(month) - 1, parseInt(day));

            return isToday(fileDate, today);

        }


        function isFileInWeek(filePath, today) {

            const fileDate = new Date(parseInt(year), parseInt(month) - 1, parseInt(day));

        // 保存任务到文件

        async function saveTaskToFile(task) {

            const fileHandle = fileHandles.get(task.file);

            if (!fileHandle) {

                throw new Error('文件不存在');

            }


            const file = await fileHandle.getFile();

            const content = await file.text();

            const lines = content.split('\n');


            if (task.isNew) {

                // 添加新任务

                const newTaskLine = formatTaskLine(task);

                lines.push(newTaskLine);

            } else {

                // 更新现有任务

                if (task.lineNumber > 0 && task.lineNumber <= lines.length) {

                    lines[task.lineNumber - 1] = formatTaskLine(task);

                }

            }


            // 写回文件

            const writable = await fileHandle.createWritable();

            await writable.write(lines.join('\n'));

            await writable.close();

        }


        // 格式化任务行

        function formatTaskLine(task) {

            let line = `- [${task.status}] ${task.text}`;

        // 执行标签替换

        async function executeTagReplace() {

            const oldTag = document.getElementById('replaceOldTag').value.trim();

            const newTag = document.getElementById('replaceNewTag').value.trim();

            const replaceMode = document.getElementById('replaceMode').checked;

            

            if (!newTag) {

                showMessage('请输入新标签', 'error');

                return;

            }

            

            if (replaceMode && !oldTag) {

                showMessage('替换模式下需要指定要替换的标签', 'error');

                return;

            }

            

            showMessage('正在修改标签...', 'info');

            

            try {

                let modifiedCount = 0;

                

                for (const taskId of selectedTasks) {

                    const [file, lineNumber] = taskId.split('_').map((part, index) => 

                        index === 1 ? parseInt(part) : part

                    );

                    

                    const task = allTasks.find(t => t.file === file && t.lineNumber === lineNumber);

                    if (!task) continue;

                    

                    // 修改标签

                    if (replaceMode) {

                        // 替换模式

                        task.tags = task.tags.map(tag => 

                            tag === oldTag ? newTag : tag

                        );

                    } else {

                        // 添加模式

                        if (!task.tags.includes(newTag)) {

                            task.tags.push(newTag);

                        }

                    }

                    

                    await saveTaskToFile(task);

                    modifiedCount++;

                }

                

                closeTagReplaceModal();

                await loadTasksInternal();

                showMessage(`成功修改 ${modifiedCount} 个任务的标签`, 'success');

                

            } catch (error) {

                showMessage('修改标签失败: ' + error.message, 'error');

            }

        }


        // 删除选中的任务

        async function deleteSelectedTasks() {

            if (selectedTasks.size === 0) {

                showMessage('请先选择要删除的任务', 'error');

                return;

            }

            

            if (!confirm(`确定要删除选中的 ${selectedTasks.size} 个任务吗？`)) {

                return;

            }

            

            showMessage('正在删除任务...', 'info');

            

            try {

                let deletedCount = 0;

                

                // 按文件分组删除

                const tasksByFile = {};

                for (const taskId of selectedTasks) {

                    const [file, lineNumber] = taskId.split('_').map((part, index) => 

                        index === 1 ? parseInt(part) : part

                    );

                    

                    if (!tasksByFile[file]) {

                        tasksByFile[file] = [];

                    }

                    tasksByFile[file].push(parseInt(lineNumber));

                }

                

                // 按文件处理删除

                for (const [file, lineNumbers] of Object.entries(tasksByFile)) {

                    const fileHandle = fileHandles.get(file);

                    if (!fileHandle) continue;

                    

                    const fileContent = await fileHandle.getFile();

                    const content = await fileContent.text();

                    let lines = content.split('\n');

                    

                    // 按行号倒序删除，避免行号偏移

                    lineNumbers.sort((a, b) => b - a);

                    

                    for (const lineNumber of lineNumbers) {

                        if (lineNumber > 0 && lineNumber <= lines.length) {

                            lines.splice(lineNumber - 1, 1);

                            deletedCount++;

                        }

                    }

                    

                    // 写回文件

                    const writable = await fileHandle.createWritable();

                    await writable.write(lines.join('\n'));

                    await writable.close();

                }

                

                selectedTasks.clear();

                updateBulkActionsUI();

                await loadTasksInternal();

                showMessage(`成功删除 ${deletedCount} 个任务`, 'success');

                

            } catch (error) {

                showMessage('删除任务失败: ' + error.message, 'error');

            }

        }

        // 渲染时间线视图

        function renderTimeline() {

            const timelineView = document.getElementById('timelineView');

            timelineView.innerHTML = '';

            

            // Group tasks by certification type

            const certTypes = {

                'developer': { name: 'Developer Cert', tasks: [] },

                'agent': { name: 'Agent Cert', tasks: [] },

                'ai': { name: 'AI-Based Cert', tasks: [] },

                'other': { name: 'Other Certification', tasks: [] }

            };

            

            // Categorize tasks

            filteredTasks.forEach(task => {

                const text = task.text.toLowerCase();

                let category = 'other';

                

                if (text.includes('developer') || text.includes('dev') || text.includes('software')) {

                    category = 'developer';

                } else if (text.includes('agent') || text.includes('broker') || text.includes('sales')) {

                    category = 'agent';

                } else if (text.includes('ai') || text.includes('machine learning') || text.includes('ml')) {

                    category = 'ai';

                }

                

                certTypes[category].tasks.push(task);

            });

            

            // Render each category

            Object.entries(certTypes).forEach(([key, category]) => {

                if (category.tasks.length === 0) return;

                

                const categoryDiv = document.createElement('div');

                categoryDiv.className = 'timeline-category';

                

                const header = document.createElement('h3');

                header.textContent = `${category.name} (${category.tasks.length})`;

                categoryDiv.appendChild(header);

                

                const tasksDiv = document.createElement('div');

                tasksDiv.className = 'timeline-tasks';

                

                // Sort tasks by effective date (due or scheduled)

                const sortedTasks = category.tasks.sort((a, b) => {

                    const aEffective = a.due || a.scheduled;

                    const bEffective = b.due || b.scheduled;

                    if (!aEffective) return 1;

                    if (!bEffective) return -1;

                    return new Date(aEffective) - new Date(bEffective);

                });

                

                sortedTasks.forEach(task => {

                    const taskDiv = document.createElement('div');

                    taskDiv.className = 'timeline-task';

                    

                    const isCompleted = task.status === 'x';

                    const effectiveDate = task.due || task.scheduled;

                    const isOverdue = effectiveDate && new Date(effectiveDate) < new Date() && !isCompleted;

                    

                    if (isCompleted) taskDiv.classList.add('completed');

                    if (isOverdue) taskDiv.classList.add('overdue');

                    

                    const headerDiv = document.createElement('div');

                    headerDiv.className = 'timeline-task-header';

                    

                    const dateDiv = document.createElement('div');

                    dateDiv.className = 'timeline-task-date';

                    if (isOverdue) dateDiv.classList.add('overdue');

                    if (effectiveDate) {

                        dateDiv.textContent = (task.due ? '📅 ' : '⏳ ') + formatDate(effectiveDate);

                    } else {

                        dateDiv.textContent = 'No date';

                    }

                    headerDiv.appendChild(dateDiv);

                    

                    const titleDiv = document.createElement('div');

                    titleDiv.className = 'timeline-task-title';

                    titleDiv.textContent = task.text;

                    headerDiv.appendChild(titleDiv);

                    

                    // Add tags inline with title

                    task.tags.forEach(tag => {

                        const tagSpan = document.createElement('span');

                        tagSpan.className = 'timeline-task-tag';

                        tagSpan.textContent = tag;

                        headerDiv.appendChild(tagSpan);

                    });

                    

                    taskDiv.appendChild(headerDiv);

                    tasksDiv.appendChild(taskDiv);

                });

                

                categoryDiv.appendChild(tasksDiv);

                timelineView.appendChild(categoryDiv);

            });

            

            if (timelineView.innerHTML === '') {

                timelineView.innerHTML = '<div class="empty-state"><h3>No tasks found</h3></div>';

            }
